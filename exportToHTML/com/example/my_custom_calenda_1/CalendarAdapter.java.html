15   /**
16    * 1. 코드 구성 및 핵심 구조 설명:
17    * 이 어댑터는 'Multi-ViewType' 전략을 사용하여 두 가지 형태의 레이아웃을 하나의 리사이클러뷰에서 관리합니다.
18    * - TYPE_DAY (0): 개별 날짜 셀
19    * - TYPE_DETAIL (1): 클릭 시 삽입되는 가로형 상세 정보 칸
20    *
21    * 사용자가 날짜를 클릭하면 해당 날짜가 포함된 행의 끝 인덱스를 계산하여 데이터 리스트(items)에
22    * 상세 정보를 삽입(notifyItemInserted)하거나 삭제(notifyItemRemoved)함으로써 동적 레이아웃 변화를 구현합니다.
23    */
24   public class CalendarAdapter extends RecyclerView.Adapter<RecyclerView.ViewHolder> {
25   
26       private static final int TYPE_DAY = 0;
27       private static final int TYPE_DETAIL = 1;
28   
29       // native) ArrayList<Object>: 다양한 타입(LocalDate 및 Event 리스트)을 하나의 리스트에서 관리하기 위한 객체 지향형 구조입니다.
30       private final ArrayList<Object> items; 
31       private final ArrayList<Event> eventList;
32       private final OnItemListener onItemListener;
33       private int detailPosition = -1;
34   
35       public interface OnItemListener {
36           void onItemClick(LocalDate date, ArrayList<Event> events, int position);
37       }
38   
39       public CalendarAdapter(ArrayList<LocalDate> dayList, ArrayList<Event> eventList, OnItemListener onItemListener) {
40           this.items = new ArrayList<>(dayList);
41           this.eventList = eventList;
42           this.onItemListener = onItemListener;
43       }
44   
45       @Override
46       public int getItemViewType(int position) {
47           // native) instanceof: 객체의 실제 타입을 확인하여 날짜인지 상세 정보인지 판별합니다.
48           if (items.get(position) instanceof LocalDate || items.get(position) == null) {
49               return TYPE_DAY;
50           } else {
51               return TYPE_DETAIL;
52           }
53       }
54   
55       @NonNull
56       @Override
57       public RecyclerView.ViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
58           // native) LayoutInflater: XML 레이아웃 파일을 실제 View 객체로 메모리에 로드하는 안드로이드 핵심 클래스입니다.
59           LayoutInflater inflater = LayoutInflater.from(parent.getContext());
60           if (viewType == TYPE_DAY) {
61               View view = inflater.inflate(R.layout.calendar_cell, parent, false);
62               return new CalendarViewHolder(view);
63           } else {
64               View view = inflater.inflate(R.layout.event_detail_item, parent, false);
65               return new DetailViewHolder(view);
66           }
67       }
68   
69       @Override
70       public void onBindViewHolder(@NonNull RecyclerView.ViewHolder holder, int position) {
71           if (holder instanceof CalendarViewHolder) {
72               CalendarViewHolder dayHolder = (CalendarViewHolder) holder;
73               LocalDate date = (LocalDate) items.get(position);
74               // native) removeAllViews: 기존에 추가되었던 형광펜 뷰들을 초기화하여 뷰 재사용 시 발생할 수 있는 문제를 방지합니다.
75               dayHolder.highlighterLayout.removeAllViews();
76   
77               if (date == null) {
78                   dayHolder.dayText.setText("");
79                   dayHolder.itemView.setOnClickListener(null);
80               } else {
81                   dayHolder.dayText.setText(String.valueOf(date.getDayOfMonth()));
82   
83                   ArrayList<Event> eventsOnDay = new ArrayList<>();
84                   for (Event event : eventList) {
85                       if (event.isWithin(date)) {
86                           eventsOnDay.add(event);
87   
88                           // native) android.view.View: 형광펜 효과를 주기 위해 빈 뷰 객체를 동적으로 생성합니다.
89                           View highlighter = new View(dayHolder.itemView.getContext());
90                           // native) LinearLayout.LayoutParams: 부모 뷰 내에서의 배치 정보(너비, 높이, 마진 등)를 설정합니다.
91                           LinearLayout.LayoutParams params = new LinearLayout.LayoutParams(
92                                   ViewGroup.LayoutParams.MATCH_PARENT, 25);
93                           params.setMargins(0, 2, 0, 2);
94                           highlighter.setLayoutParams(params);
95                           highlighter.setBackgroundColor(event.getColor());
96                           // native) addView: 생성한 형광펜 뷰를 레이아웃에 동적으로 추가합니다.
97                           dayHolder.highlighterLayout.addView(highlighter);
98                       }
99                   }
100  
101                  dayHolder.itemView.setOnClickListener(v -> {
102                      onItemListener.onItemClick(date, eventsOnDay, holder.getAdapterPosition());
103                  });
104              }
105          } else if (holder instanceof DetailViewHolder) {
106              DetailViewHolder detailHolder = (DetailViewHolder) holder;
107              ArrayList<Event> events = (ArrayList<Event>) items.get(position);
108              
109              // native) StringBuilder: 여러 문자열을 효율적으로 합치기 위한 자바 네이티브 클래스입니다.
110              StringBuilder sb = new StringBuilder();
111              for (Event e : events) {
112                  sb.append(e.getName()).append("\n");
113              }
114              detailHolder.detailTitle.setText("선택한 날짜의 일정");
115              detailHolder.detailContent.setText(sb.toString().trim());
116          }
117      }

119      /**
120       * 상세 정보 칸을 중간에 삽입/삭제하는 핵심 함수
121       */
122      public void showDetail(int dayPosition, ArrayList<Event> events) {
123          if (detailPosition != -1) {
124              // native) items.remove: 리스트에서 기존 상세 칸 데이터를 제거합니다.
125              items.remove(detailPosition);
126              // native) notifyItemRemoved: 특정 위치의 아이템이 삭제되었음을 알리고 애니메이션을 트리거합니다.
127              notifyItemRemoved(detailPosition);
128              if (dayPosition > detailPosition) {
129                  dayPosition--;
130              }
131              detailPosition = -1;
132          }
133  
134          if (events == null || events.isEmpty()) {
135              // native) notifyDataSetChanged: 리스트 전체를 다시 그리게 하여 UI를 갱신합니다.
136              notifyDataSetChanged(); 
137              return;
138          }
139  
140          int row = dayPosition / 7;
141          int insertPos = (row + 1) * 7;
142          if (insertPos > items.size()) insertPos = items.size();
143  
144          // native) items.add(index, object): 특정 위치에 데이터를 삽입합니다.
145          items.add(insertPos, events);
146          detailPosition = insertPos;
147          // native) notifyItemInserted: 새로운 아이템이 삽입되었음을 알리고 칸을 밀어내는 애니메이션을 수행합니다.
148          notifyItemInserted(detailPosition);
149      }
