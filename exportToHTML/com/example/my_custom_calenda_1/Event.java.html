1    package com.example.my_custom_calenda_1;
2    
3    import java.time.LocalDate;
4    
5    /**
6     * 1. 코드 구성 및 핵심 구조 설명:
7     * 이 클래스는 일정(Event) 정보를 관리하는 순수 Java 데이터 모델입니다.
8     * 객체 지향 프로그래밍의 캡슐화 원칙을 따라 일정명, 시작일, 종료일, 색상을 관리합니다.
9     */
10   public class Event {
11       private String name;
12       // native) java.time.LocalDate: Java 8 이상에서 제공하는 날짜 정보를 다루는 핵심 클래스입니다.
13       private LocalDate startDate;
14       private LocalDate endDate;
15       private int color;
16   
17       public Event(String name, LocalDate startDate, LocalDate endDate, int color) {
18           this.name = name;
19           this.startDate = startDate;
20           this.endDate = endDate;
21           this.color = color;
22       }
23   
24       public String getName() {
25           return name;
26       }
27   
28       public LocalDate getStartDate() {
29           return startDate;
30       }
31   
32       public LocalDate getEndDate() {
33           return endDate;
34       }
35   
36       public int getColor() {
37           return color;
38       }
39   
40       /**
41        * native) LocalDate#isEqual, isAfter, isBefore: 
42        * 특정 날짜가 일정 기간 내에 포함되는지 확인하기 위해 사용되는 네이티브 비교 함수들입니다.
43        */
44       public boolean isWithin(LocalDate date) {
45           return (date.isEqual(startDate) || date.isAfter(startDate)) &&
46                  (date.isEqual(endDate) || date.isBefore(endDate));
47       }
48   }
